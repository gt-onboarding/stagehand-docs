    interface ExtractOptions<T extends z.AnyZodObject> {
      instruction?: string;
      schema?: T;
      modelName?: AvailableModel;
      modelClientOptions?: ClientOptions;
      domSettleTimeoutMs?: number;
      selector?: string;
      iframes?: boolean;
    }

    type ExtractResult<T> = z.infer<T>;

<div id="parameters">
  ### Parameter
</div>

<ParamField path="instruction" type="string" optional>
  Beschreibung in natürlicher Sprache, welche Daten extrahiert werden sollen.
</ParamField>

<ParamField path="schema" type="z.ZodSchema | BaseModel" optional>
  Typschema, das die Struktur der zu extrahierenden Daten definiert. Gewährleistet Typsicherheit und Validierung.
</ParamField>

<ParamField path="selector" type="string" optional>
  XPath-Selektor zur Einschränkung des Extraktionsbereichs. Reduziert Tokenverbrauch und verbessert die Genauigkeit.
</ParamField>

<ParamField path="iframes" type="boolean" optional>
  Auf `true` setzen, wenn Inhalte innerhalb von iframes liegen.

  **Standard:** `false`
</ParamField>

<ParamField path="modelName" type="AvailableModel" optional>
  Standard-LLM-Modell für diese Extraktion überschreiben.
</ParamField>

<ParamField path="modelClientOptions" type="ClientOptions" optional>
  Modellspezifische Konfigurationsoptionen.
</ParamField>

<ParamField path="domSettleTimeoutMs" type="number" optional>
  Maximale Wartezeit, bis sich der DOM stabilisiert hat.

  **Standard:** `30000`
</ParamField>

<div id="response-types">
  ### Antworttypen
</div>

<Tabs>
  <Tab title="Mit Schema">
    **Rückgabe:** `Promise<ExtractResult<T>>`, wobei T Ihrem Schema entspricht

    Das zurückgegebene Objekt ist strikt gemäß Ihrer Schema-Definition typisiert.
  </Tab>

  <Tab title="Nur String">
    **Rückgabe:** `Promise<{ extraction: string }>`

    Einfache String-Extraktion ohne Schema-Validierung.
  </Tab>

  <Tab title="Keine Parameter">
    **Rückgabe:** `Promise<{ page_text: string }>`

    Unverarbeitete Darstellung des Accessibility-Baums des Seiteninhalts.
  </Tab>
</Tabs>

    <CodeGroup>
      ```typescript TypeScript
      import { z } from 'zod';

      // Schema mit Beschreibungen und Validierung
      const ProductSchema = z.object({
        price: z.number().describe("Produktpreis in USD"),
        rating: z.number().min(0).max(5).describe("Kundenbewertung auf einer Skala bis 5"),
        available: z.boolean().describe("Ob das Produkt auf Lager ist"),
        tags: z.array(z.string()).optional()
      });

      // Verschachteltes Schema
      const EcommerceSchema = z.object({
        product: z.object({
          name: z.string(),
          price: z.object({
            current: z.number(),
            original: z.number().optional()
          })
        }),
        reviews: z.array(z.object({
          rating: z.number(),
          comment: z.string()
        }))
      });
      ```

      ```python Python
      from pydantic import BaseModel, Field
      from typing import Optional, List

      # Schema mit Beschreibungen und Validierung
      class Product(BaseModel):
          price: float = Field(description="Produktpreis in USD")
          rating: float = Field(ge=0, le=5, description="Kundenbewertung auf einer Skala bis 5")
          available: bool = Field(description="Ob das Produkt auf Lager ist")
          tags: Optional[List[str]] = None

      # Verschachteltes Schema
      class Price(BaseModel):
          current: float
          original: Optional[float] = None

      class Review(BaseModel):
          rating: int
          comment: str

      class ProductDetails(BaseModel):
          name: str
          price: Price

      class EcommerceData(BaseModel):
          product: ProductDetails
          reviews: List[Review]
      ```
